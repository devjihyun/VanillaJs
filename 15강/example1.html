학습내용 - 내비게이션 UI 제작 
<script>>
window.addEventListener("load", function(){
	var depth1Li=document.querySelectorAll("nav > ul > li");

	for(var i=0; i<depth1Li.length; i++){
		depth1Li[i].addEventListener("mouseenter", function(e){
			// console.log(e.currentTarget.children);
			var link=e.currentTarget.children[0];
			link.classList.add("over");

			var sub=e.currentTarget.children[1];
			sub.style.display="block";
		});

		depth1Li[i].addEventListener("mouseleave", function(e){
			var link=e.currentTarget.children[0];
			link.classList.remove("over");

			var sub=e.currentTarget.children[1];
			sub.style.display="none";
		});

		depth1Li[i].children[0].addEventListener("focusin", function(e){ // info : nav > ul > li > a
			e.target.classList.add("over");
			var sub=e.target.nextElementSibling;
			sub.style.display="block";
		});

		var lastLiLi=depth1Li[i].children[1].children; // info : nav li li
		// console.log(lastLiLi);

		for(var j=0; j<lastLiLi.length; j++){
			if(j == (lastLiLi.length-1)){ // 4-1=3
				lastLiLi[j].addEventListener("focusout", function(e){
					// console.log("focusout event!!");
					var sub=e.currentTarget.parentElement;
					sub.style.display="none";

					var link=e.currentTarget.parentElement.parentElementSibling;
					link.classList.remove("over");
				});
			}
		}
	}

});
</script>
