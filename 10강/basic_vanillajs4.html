학습내용 
<script>
window.addEventListener("load", function(){
	var btn=document.querySelector("#queue");
	var ball=document.querySelector(".ball");
	var x=ball.offsetLeft;
	var targetX=400;

	btn.addEventListener("click", function(){
		animate1(ball, x, targetX);
	});

	function animate1(target, startx, targetx) {
		var timer=setInterval(function(){
			if(startx < targetx){
				startx+=10;
			} 
			else {
				startx=targetx;
				clearInterval(timer);
			}
			// console.log("x : "+x);
			target.style.left=startx+"px";
		}, 10);
	};
});
</script>
