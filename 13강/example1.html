학습내용 - 이벤트 대상에 인덱스 목록 추가 
<script>>
window.addEventListener("load", function(){
	var n=0;
	var menuLi=document.querySelectorAll("nav li"); // nodeList ==> 유사 배열
	var info=document.querySelectorAll(".description");

	menuLi[n].classList.add("on");
	info[n].classList.add("active");

	for(var i=0; i<menuLi.length; i++){
	menuLi[i].index=i; // 프로퍼티 추가
	menuLi[i].addEventListener("click", function(e){
		n=e.currentTarget.index;
		// console.log("n : "+n);

		// n에 해당하는 번호의 info NodeList는 actvie 클래스를 추가하고,
		// 해당하지 않는 번호의 info NodeList는 active 클래스를 제거합니다.
		for(var j=0; j<info.length; j++){
			if(j == n){
				menuLi[j].classList.add("on");
				info[j].classList.add("active");
			}
			else {
				menuLi[j].classList.remove("on");
				info[j].classList.remove("active");
			}
		}
	});
	}
});
</script>
