학습내용 - 기초 갤러리 UI 제작 
<script>
window.addEventListener("load", function(){
	var gallery=document.querySelector(".picture ul");
	// console.log(gallery.children); // A HTMLColletion
	var liList=document.querySelectorAll(".controller li");
	
	for(var i=0; i<liList.length; i++){
		liList[i].index=i;
		var n=0;

		liList[i].addEventListener("click", function(e){
			e.preventDefault(); // 이벤트를 브라우저로 전달하지 않습니다.
			n=e.currentTarget.index;
			// console.log("n : "+n);



			for(var j=0; j<liList.length; j++) {
				if(j == n){
					liList[j].classList.add("on");
					gallery.children[j].classList.add("active");
			}
			else {
				liList[j].classList.remove("on");
				gallery.children[j].classList.remove("active");
				}
			}
		});
	}
});
</script>
