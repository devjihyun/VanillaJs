<script>>
/*
#tab을 클릭하면 .gallery에 active 클래스를 추가합니다.

#tab을 클릭하면 gallery open ===> gallery close
다시 클릭하면 gallery close ===> gallery open

.gallery li에 마우스를 올리면 over 클래스를 추가하고, 내리면 over 클래스를 제거합니다.
*/

window.addEventListener("load", function(){
	var status="closed"; // 현재 상황을 지정하는 flag 변수
	var tab=document.querySelector("#tab");
	var slider=document.querySelector(".gallery")
	var sliderList=document.querySelectorAll(".gallery li");

	tab.addEventListener("click", function(e){
		if(status == "closed"){
			status="open";
			e.target.innerText="gallery close";
			slider.classList.add("active");
		}
		else {
			status="closed";
			e.target.innerText="gallery open";
			slider.classList.remove("active");
		}
		// console.log("status : "+status);
	});

	for(var i=0; i<sliderList.length; i++){
		sliderList[i].addEventListener("mouseenter", function(e){ // Bubbles No
		// sliderList[i].addEventListener("mouseover", function(e){ // Bubbles Yes
			// console.log(e.target);
			e.target.classList.add("over");
		});
		sliderList[i].addEventListener("mouseleave", function(e){
		// sliderList[i].addEventListener("mouseout", function(e){
			// console.log(e.target);
			e.target.classList.remove("over");
		});
	}

});
</script>
