Trigger the handler.
학습내용 
<script>>
window.addEventListener("load", function(){
	var target1=document.querySelector(".target1");
	var target2=document.querySelector(".target2");
	var trigger=document.querySelector(".trigger");

	target1.addEventListener("focus", function(e){
		// console.log("target1 focus event!!")
		e.target.value="target1 focus event!!"
	});
	target2.addEventListener("blur", function(e){
		// console.log("target1 blur event!!")
		e.target.value="target1 blur event!!"
	});

	trigger.addEventListener("click", function(e){
		target2.focus();
		target2.value="target2 focus event!!";
	});
});
</script>
