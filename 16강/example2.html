학습내용 - 노드 포커스 
<script>
window.addEventListener("load", function(){
	var popupOpenBtn=document.querySelector(".pop");
	var popup=document.querySelector(".layer_pop");
	var dim=document.querySelector(".transparent");
	var popupList=popup.children;

	for(var i=0; i<popupList.length; i++){
		// console.log(popupList[i]);

		if(popupList[i].className == "pop_title"){
			var closeBtn=popupList[i].children[0];
			// console.log(closeBtn);
		}
	}

	popupOpenBtn.addEventListener("click", function(e){
		e.preventDefault();
		popup.style.display="block";
		dim.style.display="block";
		closeBtn.focus();
	});
	closeBtn.addEventListener("click", function(e){
		e.preventDefault();
		popup.style.display="none";
		dim.style.display="none";
		popupOpenBtn.focus();
	});
});
</script>
